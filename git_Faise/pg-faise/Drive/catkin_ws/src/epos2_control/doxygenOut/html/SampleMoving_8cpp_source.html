77 void SampleMoving::publishSpeed(const ros::TimerEvent& event)
78 {
79  double speedMPS[2], pos_delta[2];
80  epos2_control::velocity speed;
81  geometry_msgs::Twist speedVector;
82 
83  //get velocity in meter per second
84  for (int i=left; i<=right; i++) { speedMPS[i] = sampleMovingSet.targetVelocity[i] * sampleMovingSet.maxMPS; }
85  //ROS_ERROR("[Talker-Publish] targetSpeed_left: %.4lf & targetSpeed_right: %.4lf", sampleMovingSet.targetVelocity[left], sampleMovingSet.targetVelocity[right]);
86 
87 
88  switch (sampleMovingSet.msgType) {
89  case SI: //publish current velocity in meter per second
90  speed.left=speedMPS[left];
91  speed.right=speedMPS[right];
92  sampleMovingSet.epos2MPS.publish(speed);
93  break;
94  case cmdVel: //publish current velocity in a cmd_vel message
95  speedVector.linear.x = ( speedMPS[right] + speedMPS[left] ) * 0.5;
96  speedVector.linear.y=0;
97  speedVector.linear.z=0;
98  speedVector.angular.x=0;
99  speedVector.angular.y=0;
100  speedVector.angular.z=(speedMPS[right] - speedMPS[left]) / sampleMovingSet.axisLength;
101  sampleMovingSet.epos2Twist.publish(speedVector);
102  break;
103  }
104 
105  //calculate current position depending on last position
106  for (int i=left; i<=right; i++) { pos_delta[i] = speedMPS[i] * (event.current_real - event.last_real).toSec(); }
107 
108  double polar_s =(pos_delta[right] + pos_delta[left] ) * 0.5;
109  double polar_theta=(pos_delta[right] - pos_delta[left]) / sampleMovingSet.axisLength;
110  double delta_r = (polar_theta == 0) ? polar_s : (polar_s / polar_theta) * sin(polar_theta/2) * 2;
111 
112  pos.now.x = pos.last.x + delta_r * cos(polar_theta/2 + pos.last.theta);
113  pos.now.y = pos.last.y + delta_r * sin(polar_theta/2 + pos.last.theta);
114  pos.now.theta = pos.last.theta + polar_theta;
115 
116  //copy new position to last position for next turn
117  pos.last.x = pos.now.x;
118  pos.last.y = pos.now.y;
119  pos.last.theta = pos.now.theta;
120 
121  //sending actionLib-msg for NavStack ...doesn't work by now on ros::groovy
122  /*if (targetSpeed.publishMsgType == navStack) {
123  move_base_msgs::MoveBaseGoal goal;
124 
125  goal.target_pose.header.frame_id = "base_link";
126  goal.target_pose.header.stamp = ros::Time::now();
127 
128  goal.target_pose.pose.position.x = pos.now.x;
129  goal.target_pose.pose.position.y = pos.now.y;
130  goal.target_pose.pose.orientation.w = pos.now.theta;
131 
132  //ROS_ERROR("Sending goal");
133  action.sendGoal(goal);
134  action.waitForResult();
135 
136  if(action.getState() != actionlib::SimpleClientGoalState::SUCCEEDED) {
137  ROS_ERROR("The base failed to move for some reason.");
138  }
139  }*/
140

142  sampleMovingSet.transform.setOrigin( tf::Vector3(pos.now.x, pos.now.y, 0.0) );
143  sampleMovingSet.transform.setRotation(tf::Quaternion( 0,0, sin(pos.now.theta/2), cos(pos.now.theta/2) ));

147 void SampleMoving::gamepadCallback (const sensor_msgs::Joy joymsg)
148 {
149  double targetSpeed[2] = {0,0};
150  int moving = false;
151  int lastJoyControll = sampleMovingSet.joyControll;
152  int lastGamepadMode = sampleMovingSet.gamepadMode;
153  std::string padMode="";
154  std::string gamePadType="";
155  axesStruct axesRemap;
156 
157  //init of axesRemap
158  axesRemap.leftx = 0;
159  axesRemap.lefty = 0;
160  axesRemap.rightx = 0;
161  axesRemap.righty = 0;
162  axesRemap.switchButton = 0;
163  axesRemap.quickStopButton = 0;
164 
165 //----------Switching of interfaces--------------------------//
166  switch (joymsg.axes.size()) {
167  case 2: sampleMovingSet.joyControll=wheel;
168  axesRemap.lefty = joymsg.axes[1];
169  sampleMovingSet.ableToMove = true;
170  padMode = "steering wheel with two pedals";
171  gamePadType = wheel;
172  break;
173  case 3: axesRemap.lefty=(joymsg.axes[1] - joymsg.axes[2])/2;
174  gamePadType = wheel;
175  padMode = "steering wheel with two pedals";
176  sampleMovingSet.joyControll=wheel;
177 
178  if (sampleMovingSet.ableToMove == false) {
179  padMode = "steering wheel with two pedals - but not initialed (press both pedals once)";
180  sampleMovingSet.joyControll=noJoy;
181  }
182  if (joymsg.axes[1] == -1 && joymsg.axes[2] == -1) {
183  sampleMovingSet.ableToMove = true;
184  }
185  //ROS_ERROR("xvel: %.4f, axes1: %.4f, axes2: %.4f", xVelocity, joymsg.axes[1], joymsg.axes[2]);
186  break;
187  case 4: sampleMovingSet.joyControll = gamePad;
188  gamePadType = "normal Gamepad";
189  axesRemap.leftx = joymsg.axes[0];
190  axesRemap.lefty = joymsg.axes[1];
191  axesRemap.rightx = joymsg.axes[2];
192  axesRemap.righty = joymsg.axes[3];
193  axesRemap.switchButton = joymsg.buttons[5];
194  sampleMovingSet.ableToMove = true;
195  break;
196  case 6: sampleMovingSet.joyControll = gamePad;
197  gamePadType = "Logitech F710 DirectInput";
198  axesRemap.leftx = joymsg.axes[0];
199  axesRemap.lefty = joymsg.axes[1];
200  axesRemap.rightx = joymsg.axes[2];
201  axesRemap.righty = joymsg.axes[3];
202  axesRemap.switchButton = joymsg.buttons[5];
203  axesRemap.quickStopButton = joymsg.buttons[2];
204  sampleMovingSet.ableToMove = true;
205  break;
206  case 8: sampleMovingSet.joyControll = gamePad;
207  gamePadType = "Logitech F710 XInput";
208  axesRemap.leftx = joymsg.axes[0];
209  axesRemap.lefty = joymsg.axes[1];
210  axesRemap.rightx = joymsg.axes[3];
211  axesRemap.righty = joymsg.axes[4];
212  axesRemap.switchButton = joymsg.buttons[5];
213  axesRemap.quickStopButton = joymsg.buttons[1];
214  sampleMovingSet.ableToMove = true;
215  break;
216  }
217 
218  //switching of gamepad-modes
219  if (axesRemap.leftx != 0 || axesRemap.lefty != 0 || axesRemap.rightx != 0 || axesRemap.righty != 0) {moving = true;}
220 
221  if (sampleMovingSet.joyControll == gamePad && axesRemap.switchButton == 1 && moving == false) {
222  sampleMovingSet.gamepadMode = (sampleMovingSet.gamepadMode + 1) % 3;
223 
224  switch (sampleMovingSet.gamepadMode) {
225  case oneStick: padMode = "oneStick: left Stick controls the robot (back- & forward + left & right)";
226  break;
227  case twoStick: padMode = "twoStick: left Stick back- and forward; right stick left and right";
228  break;
229  case tankStick: padMode = "tankStick: left stick = left side; right stick = right side";
230  break;
231  }
232  }
233 
234  //output new mode and interface
235  if( lastJoyControll != sampleMovingSet.joyControll || lastGamepadMode != sampleMovingSet.gamepadMode) {
236  ROS_ERROR("change humanInterface mode: (%s) %s (Count of Axis: %i)", gamePadType.c_str(), padMode.c_str(), joymsg.axes.size());
237  }
238 
239 //----------Changing of Speed depending on mode---------------//
240  if (sampleMovingSet.ableToMove == true) {
241  double Ratio; //pretend against values higher 1
242  float r_min = 0.3, r_delta = 1; //in meter - wheel
243  double sideShift = 0, radius; //wheel
244 
245  switch (sampleMovingSet.joyControll) {
246  case gamePad:
247  switch (sampleMovingSet.gamepadMode) {
248  case oneStick: //left Stick controls the robot (back- & forward + left & right)
249  Ratio = fabs(axesRemap.lefty) + fabs(axesRemap.leftx);
250  if (Ratio < 1) {Ratio = 1;}
251  targetSpeed[left] = (axesRemap.lefty - axesRemap.leftx) / Ratio;
252  targetSpeed[right] =(axesRemap.lefty + axesRemap.leftx) / Ratio;
253  break;
254  case twoStick: //left Stick back- and forward; right stick left and right
255  Ratio = fabs(axesRemap.lefty) + fabs(axesRemap.rightx);
256  if (Ratio < 1) {Ratio = 1;}
257  targetSpeed[left] = (axesRemap.lefty - axesRemap.rightx) / Ratio;
258  targetSpeed[right] =(axesRemap.lefty + axesRemap.rightx) / Ratio;
259  break;
260  case tankStick: //left stick=left side; right stick = right side
261  targetSpeed[left]=axesRemap.lefty;
262  targetSpeed[right]=axesRemap.righty;
263  break;
264  }
265  break;
266  case wheel: //left stick only, because the steering wheel got only two axes
267  if (fabs(joymsg.axes[0]) > 0.001) { //get radius and depending on it the delta between both sides
268  radius = ( r_min + r_delta * (1 - fabs(joymsg.axes[0])) ) * sign(joymsg.axes[0]);
269  sideShift = sampleMovingSet.axisLength * 0.5 / radius;
270  }
271  if ( ( fabs(axesRemap.lefty) * (1 + fabs(sideShift)) ) > 1) {//prevent against values greater 1 or less -1
272  axesRemap.lefty = 1 / (1 + fabs(sideShift)) * sign(axesRemap.lefty);
273  }
274  //ROS_ERROR("xVelocity: %.4lf & sideShift: %.4lf", xVelocity, sideShift);
275  targetSpeed[left] = axesRemap.lefty * (1 - sideShift);
276  targetSpeed[right]= axesRemap.lefty * (1 + sideShift);
277  break;
278  }//end switch joyControll
279  }//end if ableToMove
280 
281  double smoothVeloLeft = targetSpeed[left], smoothVeloRight = targetSpeed[right];
282  //smoothVelocity(targetSpeed, &smoothVeloLeft, &smoothVeloRight, sampleMovingSet.SpeedStepSize); //changes will only take effect with new joy-Messages
283  //ROS_ERROR("[gamepad] left: %.2f & right: %.2f", smoothVeloLeft, smoothVeloRight);
284 
285  sampleMovingSet.joyInUse = ( fabs(smoothVeloLeft) < 0.001 && fabs(smoothVeloRight) < 0.001 ) ? 0 : 1;
286  sampleMovingSet.targetVelocity[left] = smoothVeloLeft;
287  sampleMovingSet.targetVelocity[right] = smoothVeloRight;
288 }

378 void SampleMoving::smoothVelocity(double inputVelocity[2], double *outputVelocityLeft, double *outputVelocityRight, double speedStepSize, int useDelta)
379 {
380  double deltaDelta = 0, speedDelta = 0, currentVelocity[2];
381  int deltaSign = 0;
382  double outputVelocity[2] = {0,0};
383 
384  for (int i=left; i<=right; i++) {
385  currentVelocity[i] = sampleMovingSet.targetVelocity[i];
386  }
387 
388  deltaDelta = (inputVelocity[left] - inputVelocity[right]) - (currentVelocity[left] - currentVelocity[right]);
389  deltaDelta *= useDelta;
390 
391  if (deltaDelta == 0) {
392  for (int i=left; i<=right; i++) {
393  speedDelta = inputVelocity[i] - currentVelocity[i];
394  deltaSign = sign(speedDelta, 1);
395  outputVelocity[i] = nextToZero(deltaSign, currentVelocity[i] + deltaSign*nextToZero(speedStepSize, fabs(speedDelta)) );
396  }
397  } else {
398  deltaSign = sign(deltaDelta, 1);
399  outputVelocity[left] = nextToZero(deltaSign, currentVelocity[left] + deltaSign * nextToZero(speedStepSize, fabs(deltaDelta)/2) );
400  outputVelocity[right]= nextToZero((-1)*deltaSign, currentVelocity[right] - deltaSign * nextToZero(speedStepSize, fabs(deltaDelta)/2) );
401  }
402 
403  *outputVelocityLeft = outputVelocity[left];
404  *outputVelocityRight = outputVelocity[right];
405  //ROS_ERROR("[keyboard] left: %.2f & right: %.2f", outputVelocity[left], outputVelocity[right]);
406 }
