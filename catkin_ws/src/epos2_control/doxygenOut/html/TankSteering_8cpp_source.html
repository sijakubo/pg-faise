48 void TankSteering::driveCallback(const geometry_msgs::Twist velocityVector)
49 {
50  double dxTarget = tankSettings.axisLength * velocityVector.angular.z / 2;
51  double xTarget = velocityVector.linear.x;
52 
53  //ROS_ERROR("linear.x: %.6f ; angular.z: %.6f", velocityVector.linear.x, velocityVector.angular.z); //debugging
54 
55  if ( fabs(tankSettings.maxSafetyMPS) > 0.01 ) {
56  //forward velocity restriction
57  if ( tankSettings.maxMPS < (fabs(dxTarget) + fabs(xTarget)) ) { xTarget = (tankSettings.maxMPS - fabs(dxTarget)) * sign(xTarget); }
58 
59  double x = (xTarget > tankSettings.maxSafetyMPS) ? tankSettings.maxSafetyMPS : xTarget;
60 
61  //rotation velocity restriction
62  double dx = (fabs(dxTarget) > tankSettings.maxSafetyMPS) ? tankSettings.maxSafetyMPS * sign(dxTarget) : dxTarget;
63 
64  //ROS_ERROR("x: %.4f ; dx: %.4f", x, dx); //debugging
65 
66  //output velocities
67  tankSettings.targetVelocityMPS[left] = x - dx;
68  tankSettings.targetVelocityMPS[right] = x + dx;
69  } else {
70  ROS_ERROR("No movement possible because the maximum safety Velocity is to low.");
71  tankSettings.targetVelocityMPS[left] = 0;
72  tankSettings.targetVelocityMPS[right] = 0;
73  }

101 
102  float timeDelta = (event.current_real - event.last_real).toSec();
103  double polar_s =(pos_delta[right] + pos_delta[left] ) * 0.5;
104  double polar_theta=(pos_delta[right] - pos_delta[left]) / tankSettings.axisLength;
105  double delta_r = (polar_theta == 0) ? polar_s : (polar_s / polar_theta) * sin(polar_theta/2) * 2;
106

114  }
115 
116  pos.now.x = pos.last.x + delta_r * cos(polar_theta/2 + pos.last.theta);
117  pos.now.y = pos.last.y + delta_r * sin(polar_theta/2 + pos.last.theta);
118  pos.now.theta = pos.last.theta + polar_theta;

139  nav_odom.pose.pose.orientation = odom_quat;
140  nav_odom.twist.twist.linear.x = polar_s / timeDelta;
141  nav_odom.twist.twist.linear.y = 0;

206 int TankSteering::init()
207 {
208  int j = 0;
209  int active[2] = {0,0};
210  Epos2MotorController::EposState state = Epos2MotorController::disabled;
211 
212  for (int i = left; i <= right; i++) { tankSettings.epos[i]->testForErrorsAndPrint(); }
213  resetEPOS2Fault();
214  for (int i=left; i<=right;i++) {tankSettings.epos[i]->init(tankSettings.maxRPM, tankSettings.wheelPerimeter);}
215  ROS_INFO("initialize both EPOS2 successfully");
216 
217  j = 0;
218  do {
219  if (tankSettings.epos[left]->getEpos2State() == Epos2MotorController::enabled && tankSettings.epos[right]->getEpos2State() == Epos2MotorController::enabled) {
220  state = Epos2MotorController::enabled;
221  if (j == 0) { for (int i=left; i<=right;i++) {tankSettings.epos[i]->enableEpos2();} }
222  }
223  j++;
224  } while ( state != Epos2MotorController::enabled && j < 6);
225 
226  if (state == Epos2MotorController::enabled) {
227 
228  for (int i=left; i<=right; i++) {
229  active[i] = tankSettings.epos[i]->activateProfileVelocity();
230  }
231 
232  if ( active[left] == 1 && active[right] == 1 ) {
233 
234  tankSettings.epos2MPS = tankSettings.roshandle.subscribe("epos2_MPS_left_right", 1000, &TankSteering::driveCallbackMPS, this);
235  tankSettings.epos2Twist = tankSettings.roshandle.subscribe("cmd_vel", 1000, &TankSteering::driveCallback, this);
236 
237  for (int i=left; i<=right; i++) { pos.lastPosition[i]=tankSettings.epos[i]->getAbsolutePosition() * tankSettings.wheelPerimeter; }
238 
239  tankSettings.odomTimer = tankSettings.roshandle.createTimer(ros::Duration(tankSettings.odomDuration), &TankSteering::odomCallback, this);
240  tankSettings.setVelocity = tankSettings.roshandle.createTimer(ros::Duration(tankSettings.velocityDuration), &TankSteering::setVelocityCallback, this);
241  tankSettings.odomPub = tankSettings.roshandle.advertise<nav_msgs::Odometry>("odom", 50);
242 
243  tankSettings.getErrorLoop = tankSettings.roshandle.createTimer(ros::Duration(1.02), &TankSteering::getDeviceErrorCallback, this);
244 
245  ROS_INFO("Epos2-TankSteering is now listing for velocity messages");
246  return 1;
247 
248  } else {
249  ROS_ERROR("at least one Epos2 is not in ProfileVelocityMode - no movement is possible");
250  return 0;
251  }
252 
253  } else {
254  ROS_FATAL("Unable starting motors, the EPOS2 are not enabled. (Maybe Power-off or Unplugged?)");
255 
256  return 0;
257  }//end of state is enable

260 void TankSteering::initLaserscanner(int range, double low, double high)
261 {
262  //sicklmsCounter=0; //testing laser-scanner
263  int n=3;
264 
265  LaserDistance.low = low;
266  LaserDistance.high = high;
267  LaserDistance.range = range;
268 
269  LaserDistance.a = 2 * tankSettings.maxMPS * (n - 2) / ( n * pow((LaserDistance.low - LaserDistance.high),2) );
270  LaserDistance.b = tankSettings.maxMPS / (LaserDistance.high - LaserDistance.low) - LaserDistance.a * (LaserDistance.low + LaserDistance.high);
271  LaserDistance.c = -1 * (LaserDistance.a * pow(LaserDistance.low,2) + LaserDistance.b * LaserDistance.low);
272 
273  tankSettings.sickLMS = tankSettings.roshandle.subscribe("scan", 100, &TankSteering::LaserScanCallback, this);
274 
275  ROS_ERROR("Laser-Scanner successfully initialized (Parameter-> a: %.4lf, b: %.4lf, c: %.4lf)", LaserDistance.a, LaserDistance.b, LaserDistance.c);
